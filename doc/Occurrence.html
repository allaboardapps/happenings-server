def self.seed(happening:, dummy: true)
  fake_starts_at = Faker::Time.forward(60)
  fake_ends_at = fake_starts_at + [2, 3, 4].sample.hours

  occurrence_attrs = {
    happening: happening,
    starts_at: fake_starts_at,
    ends_at: fake_ends_at,
    archived: false,
    test: false,
    dummy: dummy
  }

  occurrence = new(occurrence_attrs)
  occurrence.save!
  occurrence
end
